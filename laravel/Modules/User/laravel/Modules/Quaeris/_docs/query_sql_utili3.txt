#questa mi fa vedere tutte le risposte totali per opzione di risposta
SELECT 
    ask_lang.answer AS answer,
    946595X2253X48241 AS value,
    COUNT(*) AS total_responses
FROM 
    lime_survey_946595
LEFT JOIN 
    lime_answers AS ask 
    ON ask.code = 946595X2253X48241
    AND ask.qid = "48241"
LEFT JOIN 
    lime_answer_l10ns AS ask_lang 
    ON ask.aid = ask_lang.aid 
    AND ask_lang.language = "it"
WHERE 
    946595X2253X48241 IS NOT NULL
    AND submitdate IS NOT NULL
GROUP BY 
    ask_lang.answer, 946595X2253X48241
ORDER BY 
    total_responses DESC;


# questa query mi fa vedere tutte le risposte di una domanda id 48241 ecc
SELECT 
    q.qid AS question_id,
    q.title AS question_title,
    a.aid AS answer_id,
    a.code AS answer_code,
    a.sortorder AS answer_order,
    a.assessment_value AS answer_value,
    al10ns.answer AS translated_answer,
    al10ns.language
FROM 
    lime_questions q
LEFT JOIN 
    lime_answers a 
    ON q.qid = a.qid
LEFT JOIN 
    lime_answer_l10ns al10ns 
    ON a.aid = al10ns.aid
WHERE 
    q.qid IN (35368)
    AND al10ns.language = 'it';
